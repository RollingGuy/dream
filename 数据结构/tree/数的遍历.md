# 先序遍历
```javascript
function preOrder(root) {
  const stack = [];
  let node = root;

  while(node || stack.length) {
    if(node) {
      console.log(node.data);
      stack.push(node);
      node = node.lchild;
    } else {
      node = stack.pop();
      node = node.rchild;
    }
  }
}

```
