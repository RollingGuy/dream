function binarySearch(arr, elem) {
  let low = 0;
  let high = arr.length - 1;
  let mid = Math.floor((high + low) / 2);
  while (low <= high) {
    if (elem < arr[mid]) {
      high = mid - 1;
    } else {
      low = mid + 1;
    }
    mid = Math.floor((high + low) / 2);
  }
  return arr[mid] === elem ? mid : -1;
}
