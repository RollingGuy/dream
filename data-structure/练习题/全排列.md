给定一个没有重复数字的序列，返回其所有可能的全排列。

```js
var permute = function(nums) {
  if (nums.length <= 1) return nums;
  const res = [];
  nums.forEach((item, index, arr) => {
    permute(arr.filter((v, i) => i !== index))
      .forEach(val => {
        if (Array.isArray(val)) {
          res.push([item, ...val]);
        } else {
          res.push([item, val]);
        }
      });
  });
  return res;
};
```
