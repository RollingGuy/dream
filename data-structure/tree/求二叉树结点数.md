# 前言
思路非常简单，遍历整棵树就可以了。

二叉树数据结构：
```ts
class BTree<T> {
  data: T;
  left?: BTree<T>;
  right?: BTree<T>;
}
type T = string;
```

# 递归
```ts
function countBinaryTreeNodes(tree: BTree<T>): void {
  let counter = 0;
  const traverse = (node: BTress<T>) => {
    if (!node) return;
    counter ++;
    countBinaryTreeNodes(tree.left);
    countBinaryTreeNodes(tree.right);
  };
  traverse(tree);
  return counter;
}
```

# 非递归

## 先序遍历
```ts
function countBinaryTreeNodes(tree: BTree<T>): number {
  let counter: number = 0;
  let node: BTree<T> = tree;
  const stack: BTree<T>[] = [];
  while (node || stack.length) {
    if (node) {
      stack.push(node);
      node = node.left;
      counter ++;
    } else {
      node = stack.pop();
      node = node.right;
    }
  }
  return counter;
}
```

## 后序遍历
```ts
function countBinaryTreeNodes(tree: BTree<T>): number {
  let counter = 0;
  let node: BTree<T> = tree;
  let prev: BTree<T>;
  const stack: BTree<T>[] = [];

  while (node || stack.length) {
    while (node) {
      stack.push(node);
      node = node.left;
    }

    if (stack.length) {
      node = stack[stack.length - 1];
      if (!node.right || node.right === prev) {
        prev = node = stack.pop();
        counter ++;
        node = null;
      } else {
        node = node.right;
      }
    }
  }
  
  return counter;
}
```
