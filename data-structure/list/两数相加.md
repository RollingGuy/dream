/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function (l1, l2) {
  let num1 = '';
  let num2 = '';
  while (l1 !== null) {
    num1 = l1.val + num1;
    l1 = l1.next;
  }
  while (l2 !== null) {
    num2 = l2.val + num2;
    l2 = l2.next;
  }
  let p;
  let l3 = {};
  p = l3;
  String(BigInt(num1) + BigInt(num2)).split('').reverse().forEach(num => {
    l3.next = new ListNode(num);
    l3 = l3.next;
  });
  return p.next;
};
