思路：一开始想的是，将另一个子树翻转之后，然后比较二者的遍历结果，但是会遇到结构不同，但是val相同的情况，所以在子节点不存在的时候加一个占位符即可

```ts
type TNode = {
  val: string | number;
  left: TNode | null;
  right: TNode | null;
}
function isSymmetrical(root) {
  // 空二叉树被认为是镜像的
  if (!root) return true
  const res0 = traverse(root.left)
  shift(root.right) // 翻转
  const res1 = traverse(root.right)
  return res0.every((val, index) => val === res1[index])
}
function traverse(root) {
  const res = []
  const stack = []
  let node = root
  while (node || stack.length) {
    if (node) {
      stack.push(node)
      res.push(node.val)
      node = node.left
    } else {
      node = stack.pop()
      node = node.right
      res.push('#') // 防止val相同，结构不同的情况
    }
  }
  return res
}
function shift(root) {
  if (!root) return
  const temp = root.left
  root.left = root.right
  root.right = temp
  shift(root.left)
  shift(root.right)
}
```
