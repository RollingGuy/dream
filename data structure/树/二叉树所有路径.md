# 求二叉树的所有路径
```js
function getAllPath(tree, path = []) {
  if (!tree) return
  path.push(tree.data)
  if (!tree.left && !tree.right) {
    console.log(path.join(','))
    return
  }
  if (tree.left) {
    getAllPath(tree.left, [...path])
  }
  if (tree.right) {
    getAllPath(tree.right, [...path])
  }
}
```
