# Vue响应式原理

## 从initData开始
`src/intance/state.js`

```javascript
function initData (vm) {
  let data = vm.$options.data;
  data = vm._data = typeof data === 'function'
    ? getData(data, vm) // vm绑定为data函数的this，获取data对象，出错返回{}
    : data || {};
  
  if (!isPlainObject(data)) {
    data = {}
  }
  // proxy data on instance
  const keys = Object.keys(data)
  let i = keys.length
  while (i--) {
    const key = keys[i]
    // 如果不是保留key， 具体是通过判断key是否是 _|$开头的
    if (!isReserved(key)) {
      // 将this._data[key] 代理到 this[key]上
      proxy(vm, `_data`, key)
    }
  }
  // observe data
  observe(data, true /* asRootData */)
}

// 将this[sourceKey][key] 代理到 target[key]上
function proxy (target, sourceKey, key) {
  sharedPropertyDefinition.get = function proxyGetter () {
    return this[sourceKey][key]
  }
  sharedPropertyDefinition.set = function proxySetter (val) {
    this[sourceKey][key] = val
  }
  Object.defineProperty(target, key, sharedPropertyDefinition)
}

```
> 上面删掉了开发环境下的一些提示信息，比如data和methods、props不能有重名属性，他们的优先级为：`props > methods > data`

所以initData做了一下事情：
+ 获取data
+ 将_data上的属性代理到this上
+ 将data变为响应式的

## observe
上面讲到`observe(data, true /* asRootData */);`

```javascript
/** 
 * observe有两个参数，分别是要观测的数据、是否是根级数据的一个布尔值
*/
function observe (value, asRootData) {
  // 如果说虚拟DOM或者不是Object
  if (!isObject(value) || value instanceof VNode) {
    return
  }
  let ob
  if (hasOwn(value, '__ob__') && value.__ob__ instanceof Observer) {  // 已经被观测
    ob = value.__ob__
  } else if ( // _isVue判断是否为Vue的实例对象，避免观测Vue的实例对象
    shouldObserve &&
    !isServerRendering() &&
    (Array.isArray(value) || isPlainObject(value)) &&
    Object.isExtensible(value) &&
    !value._isVue
  ) {
    ob = new Observer(value)
  }
  if (asRootData && ob) {
    ob.vmCount++
  }
  return ob
}


// 将value保存到实例的value属性中，并且新添一个__ob__，表示已被观察
export class Observer {

  constructor (value) {
    this.value = value
    this.dep = new Dep()
    this.vmCount = 0
    def(value, '__ob__', this)

    /** 
     * 经过以上操作之后 Observer的实例对象就是：
     * this = {
     *  value: value, // value.__ob__ = this;
     *  dep: new Dep(),
     *  vmCount: 0,
     * }
    */

    if (Array.isArray(value)) {
      if (hasProto) {
        protoAugment(value, arrayMethods)
      } else {
        copyAugment(value, arrayMethods, arrayKeys)
      }
      this.observeArray(value)
    } else {
      this.walk(value)
    }
  }

  walk (obj) {
    const keys = Object.keys(obj)
    for (let i = 0; i < keys.length; i++) {
      defineReactive(obj, keys[i])
    }
  }
  
  observeArray (items) {
    for (let i = 0, l = items.length; i < l; i++) {
      observe(items[i])
    }
  }
}

/** 
  obj: Object,
  key: string,
  val: any,
  customSetter?: ?Function,
  shallow?: boolean
*/
function defineReactive (obj, key, val, customSetter, shallow) {
  const dep = new Dep()

  const property = Object.getOwnPropertyDescriptor(obj, key)
  if (property && property.configurable === false) {
    return
  }

  const getter = property && property.get
  const setter = property && property.set
  // 如果一个对象本身已经存在getter，则Vue不会对其进行深度观测，
  // 但是，经过defineReactive，这个属性就有了getter和setter，重新赋值一个对象的话  这个属性就又会被深度观测
  // 所以Vue这里进行了判断，如果一个属性存在setter，即使有getter也要对其进行深度观测
  if ((!getter || setter) && arguments.length === 2) {
    val = obj[key]
  }
  // observe就是深度观测val，防止如果obj[key]是对象的话，也得让这个对象变成响应式的
  // 这个chilOb就是当前属性本身的__ob__
  let childOb = !shallow && observe(val)
  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get: function reactiveGetter () {
      const value = getter ? getter.call(obj) : val
      // 这里大概就是某个地方如果需要在set这个属性的时候更新（callback）
      // 就把 Dep.target = callback， 然后手动obj[key]
      // 这样就会触发现在这个getter，判断Dep.target非空，于是就将这个依赖收集
      if (Dep.target) {
        dep.depend()
        if (childOb) {  // 当前这个属性的值为object|array childOb才不为空，则说明可以给这个对象上添加属性的，所以需要再收集一份相同依赖到childOb.dep中。
          childOb.dep.depend()  // 不管是谁的dep.depend，执行的操作都是将Dep.target收集起来
          if (Array.isArray(value)) {
            dependArray(value)
          }
        }
      }
      return value
    },
    set: function reactiveSetter (newVal) {
      const value = getter ? getter.call(obj) : val
      if (newVal === value || (newVal !== newVal && value !== value)) {
        return
      }

      if (getter && !setter) return
      if (setter) { // 如果有setter，则通过setter去更新原先的值，否则直接赋值
        setter.call(obj, newVal)
      } else {
        val = newVal
      }
      childOb = !shallow && observe(newVal) // 更新childOb
      // 触发依赖的cb
      dep.notify()
    }
  })
}
```

上面中defineReactive中的dep和chilOb.dep的区别是什么？<br />
首先明确一点，就是两者收集的依赖是同一个callback，只不过两者的执行时机不同。
前者是中set函数中触发，而后者是在使用`$set`或者`Vue.set`给数据对象添加新属性的时候触发。

`dep.notify(); `简单理解就是更新数据，但是如果没有`chilOb.dep`，那么只能是修改原对象属性的值的时候才会触发依赖，在新添加属性的时候却不会触发依赖，但是给某个对象新添属性的确是修改了对象。

所以在Vue中，有一个set方法：
```javascript
Vue.set = function set (target, key, val) {
  if (Array.isArray(target) && isValidArrayIndex(key)) {
    target.length = Math.max(target.length, key)
    target.splice(key, 1, val)
    return val
  }
  if (key in target && !(key in Object.prototype)) {
    target[key] = val
    return val
  }
  const ob = target.__ob__
  if (target._isVue || (ob && ob.vmCount)) {
    return val
  }
  if (!ob) {
    target[key] = val
    return val
  }
  defineReactive(ob.value, key, val)
  ob.dep.notify()
  return val
};
```

## 对数组的处理
回到上面对数组的判断：
```javascript
const arrayProto = Array.prototype
const arrayMethods = Object.create(arrayProto)
const arrayKeys = Object.getOwnPropertyNames(arrayMethods)
// Vue在arrayMethods上定义了7个方法，具体代码看下面， 所以arrayKeys就是那可以改变原数组的七个方法名字

if (Array.isArray(value)) {
  if (hasProto) { // 判断是否支持__proto__属性
    protoAugment(value, arrayMethods)
  } else {
    copyAugment(value, arrayMethods, arrayKeys)
  }
  this.observeArray(value)
}
```

// 
```javascript
function protoAugment (target, src) {
  target.__proto__ = src
}

function copyAugment (target src, keys) {
  for (let i = 0, l = keys.length; i < l; i++) {
    const key = keys[i]
    def(target, key, src[key])
  }
}
```

> Vue通过改写数组的7个属性，来实现对数组的监测，因为Object.defineProperty对数组不起作用，所有才使用这样的方法。

具体的代码如下：
```javascript
const methodsToPatch = [
  'push',
  'pop',
  'shift',
  'unshift',
  'splice',
  'sort',
  'reverse'
]

methodsToPatch.forEach(function (method) {
  const original = arrayProto[method] //原声方法
  def(arrayMethods, method, function mutator (...args) {
    const result = original.apply(this, args)
    const ob = this.__ob__
    let inserted
    switch (method) {
      case 'push':
      case 'unshift':
        inserted = args
        break
      case 'splice':
        inserted = args.slice(2)
        break
    }
    // 上面的switch就是判断是否新加进来新元素，如果有，就对这个新元素进行observe，然后触发依赖
    if (inserted) ob.observeArray(inserted)
    ob.dep.notify()
    return result
  })
})

```

> 总结一下对数组的处理：改写7个方法，首先执行原声方法，然后判断是否新添元素，如果有，对新加元素进行observe，然后触发依赖即可。
